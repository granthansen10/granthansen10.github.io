library(cfbfastR)
library(tidyverse)
library(ggrepel)
library(ggalt)
library(waffle)

plays <- load_cfb_pbp(2022)

fumbles<-plays %>% filter(offense_play == "Nebraska", grepl("Fumble", play_type)) 

nebraskaplays<-plays %>% 
  filter(offense_play == "Nebraska") %>%
  mutate(
    run_pass = case_when(
      grepl("Pass", play_type) == TRUE ~ "Pass",
      play_type == "Sack" ~ "Pass",
      grepl("Interception", play_type) == TRUE ~ "Pass",
      grepl("Rush", play_type) == TRUE ~ "Run",
      TRUE ~ "Other"
    )) %>% group_by(run_pass) %>%
  tally()

whipthirdshort<-plays %>% 
  filter(offense_play == "Nebraska", down == "3" | down=="4", distance< 4 ) %>%
  mutate(
    run_pass = case_when(
      grepl("Pass", play_type) == TRUE ~ "Pass",
      play_type == "Sack" ~ "Pass",
      grepl("Interception", play_type) == TRUE ~ "Pass",
      grepl("Rush", play_type) == TRUE ~ "Run",
      TRUE ~ "Other"
    )) %>% group_by(run_pass) %>%
  tally()


nulateshort<-c("Rushing Plays"=62, "Passing Plays"= 38.3)

nuweektoweek<-plays %>% 
  filter(offense_play == "Nebraska") %>% 
  mutate(
    run_pass = case_when(
      grepl("Pass", play_type) == TRUE ~ "Pass",
      play_type == "Sack" ~ "Pass",
      grepl("Interception", play_type) == TRUE ~ "Pass",
      grepl("Rush", play_type) == TRUE ~ "Run",
      TRUE ~ "Other",
    )) %>% group_by(week, run_pass, def_pos_team) %>% filter(run_pass != "Other")%>%  mutate(week = case_when(
    def_pos_team == "Northwestern" ~ 0,
    TRUE ~ as.numeric(week)
  )) %>% 
  tally() 

ggplot()+
  geom_bar(data = nuweektoweek, aes(x=reorder(def_pos_team, week), weight=n, fill=run_pass)) +
  scale_fill_manual(values = c("black", "red"), name= "Play Type")+
    labs(
    x="", 
    y="Total Plays", 
    title="Nebraska was balanced almost all season", 
    subtitle="Scott Frost was fired following the loss to Georgia Southern. But, there is little to no change\n in the run-pass ratio once Frost is off the board.", 
    caption="Source: cfbfastR | By: Grant Hansen") +
    theme_minimal()+
      theme(
    plot.title = element_text(size = 18, face= "bold"),
    axis.title = element_text(size=10),
    plot.subtitle = element_text(size=10),
    plot.title.position = "plot",
    panel.grid.minor = element_blank())+
  coord_flip()

metrics<-cfbd_metrics_ppa_teams(2022) %>% 
  filter(conference=="Big Ten") %>% 
  mutate(gap=as.numeric(off_passing)-as.numeric(off_rushing))

ggplot() +
  geom_dumbbell(
    data = metrics,
    aes(y=reorder(team, (gap)), x=as.numeric(off_rushing), xend=as.numeric(off_passing)), 
    size= 2.5,
    color="grey",
    colour_x="red",
    colour_xend= "black") +  
    labs(
    x="Projected Points Added", 
    y="", 
    title="Nebraska's run-pass ratio was balanced,\n its offensive efficency wasn't", 
    subtitle="Red dots equal projected points added from rushing plays. Black dots represent passing plays.", 
    caption="Source: cfbfastR | By: Grant Hansen")+
  theme_minimal()+
